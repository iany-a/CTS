Builder

public class Outer{
    //private String name; //inherited
    private String desc;
    private int value;
    
    protected Outer(String name){
        super(name);
    }

    //public getters
    //no setters

    public static class InnerBuilder{
        Outer outer;

        public InnerBuilder(String name){
            this.Outer.outer = new Outer(name);
        }


        public InnerBuilder addDesc(String desc){
            this.Outer.desc = desc;
            return this;
        }

        public InnerBuilder addValue(int value){
            this.Outer.value = value;
            return this;
        }

        //build() - last step is called from Outer class
        public Outer build(){
            return this.outer;
        }
    }
}

public class ModelFactory{
    public static AbstractModel getModel(ModelType type, String name){
        return switch(type){
            case MODEL1 -> new Model1(name, 1);
            case MODEL2 -> new Model2(name, "B");
            default -> throw new UnsupportedOperationException("Invalid model");
        }
    }
